\newcommand{\EduWFTwoDiagram}{%

  \node (error) [edu process, right=40mm of valid] {Return the validation error and stop organization creation.};

\newcommand{\EduWFFourDiagram}{%
\begin{tikzpicture}[edu activity]
  \node (start) [edu startstop] {Start};
  \node (controls) [edu process, below=of start] {Owner, teacher, or student opens the workspace controls.};
  \node (action) [edu decision, below=of controls] {Switch organization?};
  \node (orgselect) [edu process, right=40mm of action] {Select a different organization from the organization switcher.};
  \node (orgrefresh) [edu process, below=16mm of action] {Set the default organization and refresh memberships, classes, users, and live-session state.};
  \node (roleselect) [edu process, left=40mm of action] {Select a different active organization role from the role switcher.};
  \node (rolevalid) [edu decision, below=of roleselect] {Requested role active for the current organization membership?};
  \node (reject) [edu process, left=40mm of rolevalid] {Reject the role change and keep the current role state.};
  \node (routevalid) [edu decision, below=24mm of orgrefresh] {Current route still valid after the switch?};
  \node (stay) [edu process, below=of routevalid] {Remain in the current route and continue with the refreshed context.};
  \node (dashboard) [edu process, right=40mm of routevalid] {Redirect the user back to the dashboard.};
  \node (end) [edu startstop, below=of stay] {End};

  \path [edu arrow] (start) -- (controls);
  \path [edu arrow] (controls) -- (action);
  \path [edu arrow] (action.east) -- node[edu label] {Yes} (orgselect.west);

  \path [edu arrow] (roleselect) -- (rolevalid);
  \path [edu arrow] (rolevalid.east) |- node[edu label, pos=0.2] {Yes} (routevalid.west);
  \path [edu arrow] (rolevalid.west) -- node[edu label] {No} (reject.east);
  \path [edu arrow] (orgrefresh) -- (routevalid);
  \path [edu arrow] (routevalid) -- node[edu label] {Yes} (stay);
